SELECT N,'Root' T,BST.P from BST
WHERE
P IS NULL;
SELECT DISTINCT BST.N,'Inner' T,BST.P from BST
WHERE
P IS NOT NULL
AND BST.N IN (SELECT DISTINCT P FROM BST);
SELECT DISTINCT BST.N,'Leaf' T, P from BST
WHERE
BST.P IS NOT NULL
MINUS
SELECT DISTINCT BST.N,'Leaf' T, P from BST
WHERE
BST.P IS NOT NULL
AND BST.N IN (SELECT DISTINCT P FROM BST);
